## 优雅应对复杂业务场景

### 显示与隐藏

```jsx
<Form>
  <FormItem name="showDep" label="showDep" defaultValue={true}>
    <Radio.Group>
      <Radio value={true}>显示</Radio>
      <Radio value={false}>隐藏</Radio>
    </Radio.Group>
  </FormItem>
  <FormItem name="show" label="show" show={(core, { $showDep }) => !!$showDep}>
    <Input />
  </FormItem>
  <FormItem name="other" label="other">
    <Input />
  </FormItem>
</Form>
```

<Form>
  <FormItem name="showDep" label="showDep" defaultValue={true}>
    <Radio.Group
      options={[
        { label: "显示", value: true },

      ]}
    />
  </FormItem>

    name="show"
    label="show"
    show={(core, { $showDep }) => {
      return !!$showDep;

    <Input />
  </FormItem>
</Form>

### 视图控制

```jsx
<Form>
  <FormItem name="viewDep" label="viewDep">
    <Input />
  </FormItem>
  <FormItem
    name="view"
    label="view"
    view={(core, { $viewDep }) => {
      if ($viewDep === "input") {
        return (
          <FormItem name="view">
            <Input />
          </FormItem>
        );
      } else {
        return '当viewDep的值为"Input"时，为Input';
      }
    }}
  />
  <FormItem name="other" label="other">
    <Input />
  </FormItem>
</Form>
```

### 动态校验

```jsx
<Form>
  <FormItem name="validateDep" label="validateDep" defaultValue={true}>
    <Radio.Group
      options={[
        { label: "校验", value: true },
        { label: "不校验", value: false },
      ]}
    />
  </FormItem>
  <FormItem
    name="validate"
    label="validate"
    validateConfig={(val, core) => ({ required: val.validateDep })}
  >
    <Input />
  </FormItem>
  <FormItem name="other" label="other">
    <Input />
  </FormItem>
  <FormItem
    view={(core) => <Button onClick={() => core.validate()}>校验</Button>}
  />
</Form>
```

<Form>

      ]}
    />
  </FormItem>

</Form>

### 自增 List

```jsx
<Form>
  <FormItem name="arrayList" label="arrayList">
    <ArrayList
      rowKey="id"
      bottom={
        <ArrayAction>
          {({ addBottom, addTop }) => (
            <div>
              <Button type="primary" onClick={() => addBottom()}>
                底部添加
              </Button>
              <Button type="primary" onClick={() => addTop()}>
                顶部添加
              </Button>
            </div>
          )}
        </ArrayAction>
      }
    >
      <h3>任何内容</h3>
      <FormItem name="name" label="姓名">
        <Input />
      </FormItem>
      <FormItem
        view={(core, { $id }) => (
          <ArrayAction>
            {({ insertBefore, insertAfter, remove }) => {
              return (
                <div>
                  <Button
                    onClick={() => {
                      insertBefore($id);
                    }}
                  >
                    顶部增加
                  </Button>
                  <Button
                    onClick={() => {
                      insertAfter($id);
                    }}
                  >
                    底部增加
                  </Button>
                  <Button
                    onClick={() => {
                      remove($id);
                    }}
                  >
                    删除
                  </Button>
                </div>
              );
            }}
          </ArrayAction>
        )}
      />
    </ArrayList>
  </FormItem>
</Form>
```

### 自增 Table

```jsx
<Form>
  <FormItem name="arrayTable" label="arrayTable">
    <ArrayTable
      bottom={
        <ArrayAction>
          {({ addBottom, addTop }) => {
            return (
              <div>
                <Button
                  type="primary"
                  onClick={() => {
                    addBottom();
                  }}
                >
                  底部添加
                </Button>
                <Button
                  type="primary"
                  onClick={() => {
                    addTop();
                  }}
                >
                  顶部添加
                </Button>
              </div>
            );
          }}
        </ArrayAction>
      }
      tableConfig={{
        rowKey: "id",
        columns: [
          {
            dataIndex: "name",
            title: "姓名",
            render: () => {
              return (
                <FormItem name="name">
                  <Input />
                </FormItem>
              );
            },
          },
          {
            dataIndex: "opr",
            title: "操作",
            render: (t, v) => (
              <ArrayAction>
                {({ remove, insertAfter, insertBefore }) => {
                  return (
                    <div>
                      <Button
                        onClick={() => {
                          insertAfter(v.id);
                        }}
                      >
                        底部增加
                      </Button>
                      <Button
                        onClick={() => {
                          insertBefore(v.id);
                        }}
                      >
                        顶部增加
                      </Button>
                      <Button
                        onClick={() => {
                          remove(v.id);
                        }}
                      >
                        删除
                      </Button>
                    </div>
                  );
                }}
              </ArrayAction>
            ),
          },
        ],
      }}
    />
  </FormItem>
</Form>
```
